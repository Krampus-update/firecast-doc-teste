Os exemplos acima são arquivos de definições de frame utilizados em tags layout da seguinte forma em um Lua Form:

 

\<?xml version="1.0" encoding="UTF-8"?> \<form name="frmFichaTeste">                  \<layout align="client" frameStyle="/frames/frame1.xml"                  margins="{left=10, top=10, right=10, bottom=10}">                  \<label align="client" horzTextAlign="center" text="Conteudo do Layout"/>                        \</layout> \</form>
